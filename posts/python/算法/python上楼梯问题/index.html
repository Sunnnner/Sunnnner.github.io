def demo(n):                  
    if n == 1:                
        return 1              
    if n == 2:                
        return 2              
    a,b = 1, 2                
    result = 0                
    for i in range(3, n+1):   
        result = a + b        
        a = b                 
        b = result            
    return result             
                              
print(demo(10))
